9    def set_snapshot_name():
10       """ Функция создания имени файла 
11       Генерирует имя файл из текущей даты и времени 
12       """
13       # TODO: оформить парсинг в красивом виде
14   
15       ts = datetime.datetime.now()
16       file_name = str(ts.year) + \
17                   "-" + \
18                   str(ts.month) + \
19                   "-" + \
20                   str(ts.day) + \
21                   "-" + \
22                   str(ts.hour) + \
23                   "-" + \
24                   str(ts.minute) + \
25                   "-" + \
26                   str(ts.second) + \
27                   ".jpg"
28       return file_name

31   class Camera:
32       """ Базовый класс для камер 
33    
34       :param ip: IP адрес камеры (default 192.168.0.20) 
35       """
36   
37       def __init__(self, ip='192.168.0.20'):
38           self.ip = ip
39   
40           # чтение и загрузка конфигурации из файла
41           config = configparser.ConfigParser()
42           config.read("D:/Diploma/Detection/settings.ini")
43   
44           self.data_dir = config["CAMERA"]["DATA_DIR"]
45           os.chdir(self.data_dir)
46   
47       @property
48       def check_connection(self):
49           """ Функция проверки соединения 
50    
51           """
52           try:
53               response = requests.get("http://" + self.ip)
54               if response.status_code == 200:
55                   print(f"Camera with {self.ip} is online.")
56               return True
57           except requests.ConnectionError:
58               print(f"Camera with {self.ip} is offline")
59               return False
60   
61       def get_snapshot(self):
62           """ Функция захвата картинки и сохраниния файла с текущим таймстампом 
63    
64           """
65   
66           file_snapshot_name = set_snapshot_name()
67           url = 'http://' + self.ip + '/jpeg/jpeg.jpg'
68           r = requests.get(url)
69           with open(self.data_dir + '\\' + file_snapshot_name, 'wb') as f:
70               f.write(r.content)
71           return file_snapshot_name
